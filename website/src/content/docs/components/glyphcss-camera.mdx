description: Perspective or orthographic camera handle shared by the renderer and hit layer.

import { Tabs, TabItem } from '@astrojs/starlight/components';

`<GlyphcssCamera>` (alias for `<GlyphcssPerspectiveCamera>`) is the canonical projection
state. The renderer and the hit layer both read from it — mutating `rotY` updates
both simultaneously.

## Props — perspective

| Prop | Type | Default | Description |
|---|---|---|---|
| `rotX` | `number` (radians) | `0` | Tilt around X axis |
| `rotY` | `number` (radians) | `0` | Spin around Y axis |
| `distance` | `number` | `3` | Camera distance — smaller = more perspective foreshortening |
| `zoom` | `number` | `0.4` | Camera zoom — mesh fills this fraction of `min(cols, rows)` |
| `stretch` | `number` | `1.0` | Extra X scale on top of `cellAspect` |

## Props — orthographic

| Prop | Type | Default | Description |
|---|---|---|---|
| `rotX` | `number` (radians) | `0` | Tilt around X axis |
| `rotY` | `number` (radians) | `0` | Spin around Y axis |
| `zoom` | `number` | `0.4` | Scale (fraction of `min(cols, rows)`). Replaces `distance`. |

import {

  GlyphcssPerspectiveCamera,
  GlyphcssOrthographicCamera,
  GlyphcssMesh,

const octa = octahedronPolygons({ center: [0, 0, 0], size: 1, color: "#ffcc44" });

// Perspective (default)
export function PerspectiveExample() {
  return (
    <GlyphcssScene mode="solid" cols={100} rows={30}>
      <GlyphcssPerspectiveCamera rotX={0.38} zoom={0.4} distance={3} stretch={0.95} />
      <GlyphcssOrbitControls />
      <GlyphcssMesh polygons={octa} />
    </GlyphcssScene>
  );
}

// Orthographic — parallel lines stay parallel (isometric style)

  return (
    <GlyphcssScene mode="solid" cols={100} rows={30}>

      <GlyphcssOrbitControls />
      <GlyphcssMesh polygons={octa} />
    </GlyphcssScene>
  );
}

  <TabItem label="Vanilla JS">

```ts
import {
  createGlyphcssScene,
  createGlyphcssPerspectiveCamera,
  createGlyphcssOrthographicCamera,
} from "glyphcss";
import { octahedronPolygons } from "@glyphcss/core";

const host = document.querySelector<HTMLElement>("#scene")!;
const scene = createGlyphcssScene(host, { mode: "solid" });

// Perspective camera
const perspective = createGlyphcssPerspectiveCamera({
  rotX: 0.38,
  zoom: 0.4,
  distance: 3,
  stretch: 0.95,
});
scene.setOptions({ camera: perspective });

// Or switch to orthographic:
const ortho = createGlyphcssOrthographicCamera({ rotX: 0.62, zoom: 0.4 });
scene.setOptions({ camera: ortho });

scene.add(octahedronPolygons({ center: [0, 0, 0], size: 1, color: "#ffcc44" }));

// Mutate camera state and re-render:
perspective.rotY = Math.PI / 4;
perspective.distance = 4;
scene.rerender();
```

## Notes

- Distance ranges that work well: `1.6` (strong perspective) → `5.0` (nearly flat).
- `rotX: 0` looks straight down the Z axis. Try `0.38` for a slight downward tilt
  that gives the mesh visual weight.
- `stretch: 0.95` counteracts over-stretching from `cellAspect ≈ 2` on typical
  monospace fonts. Leave at `1.0` if you tighten `line-height` instead.
- Orthographic cameras ignore `distance` — use `zoom` to scale the mesh instead.
