## Geometry primitives

```ts
type Vec2 = [number, number];
type Vec3 = [number, number, number];

interface TextureTriangle {
  vertices: [Vec3, Vec3, Vec3];
  uvs?: [Vec2, Vec2, Vec2];
}

type EdgeWeight = 1 | 2 | 3; // thin / normal / core

interface WireframeEdge {
  from: Vec3;
  to: Vec3;
  weight?: EdgeWeight;
}
```

## Lighting

```ts
interface GlyphcssDirectionalLight {

interface GlyphcssAmbientLight {

}
```

## Camera

```ts
interface GlyphcssPerspectiveCameraOptions {
  rotX?: number;
  rotY?: number;
  distance?: number;
  scale?: number;
  stretch?: number;
  center?: [number, number]; // normalized [0..1, 0..1]
}

interface GlyphcssOrthographicCameraOptions {
  rotX?: number;
  rotY?: number;
  zoom?: number;
  center?: [number, number];
}

interface GlyphcssCamera {
  readonly kind: "perspective" | "orthographic";
  rotX: number; rotY: number;
  distance: number; scale: number; stretch: number;
  project(v: Vec3, cols: number, rows: number, cellAspect: number): [number, number, number];
}

type GlyphcssCameraOptions = GlyphcssPerspectiveCameraOptions | GlyphcssOrthographicCameraOptions;
type GlyphcssPerspectiveCameraHandle = GlyphcssCamera;
type GlyphcssOrthographicCameraHandle = GlyphcssCamera;
```

## Scene

```ts
interface GridSize {
  cols: number;
  rows: number;
  cellAspect: number; // cellH / cellW — typically ~2.0 for monospace
}

type RenderMode = "wireframe" | "solid" | "voxel";

interface SceneContextOptions {
  camera: GlyphcssCamera;
  grid: GridSize;
  triangles?: TextureTriangle[];
  wireframe?: WireframeEdge[];
  mode?: RenderMode;
  directionalLight?: GlyphcssDirectionalLight;
  ambientLight?: GlyphcssAmbientLight;
}

interface SceneContext {
  camera: GlyphcssCamera;
  grid: GridSize;
  triangles: TextureTriangle[];
  wireframe: WireframeEdge[];
  mode: RenderMode;
  directionalLight: GlyphcssDirectionalLight;
  ambientLight: GlyphcssAmbientLight;
}
```

## Hotspots

```ts
interface Hotspot {
  id: string;
  at: Vec3;
  size?: [number, number]; // cols × rows in character cells; default [1, 1]
}

interface HotspotCell {
  id: string;
  col: number;
  row: number;
  depth: number; // camera-space Z
  visible: boolean; // false when behind camera or off-grid
}
```
