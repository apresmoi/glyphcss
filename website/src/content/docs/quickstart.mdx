## Hello cuboctahedron

<Tabs syncKey="fw">
  <TabItem label="React">

```tsx
import { GlyphCamera, GlyphScene, GlyphOrbitControls, GlyphMesh, GlyphHotspot } from "@glyphcss/react";

export function App() {
  return (
    <GlyphCamera rotX={0.4} zoom={0.32}>
      <GlyphScene>
        <GlyphOrbitControls drag wheel />
        <GlyphMesh geometry="cuboctahedron">
          <GlyphHotspot at={[1, 0, 0]} onClick={() => alert("vertex")} />
        </GlyphMesh>
      </GlyphScene>
    </GlyphCamera>
  );
}
```

## What you get

- A single `<pre>` rendered as ASCII glyphs, animating a full rotation in ~6 seconds.
- One absolutely-positioned `<div role="button">` for each `<GlyphHotspot>`,
  tracking the rotating mesh exactly (per-frame keyframes synced via `steps(N)`).
- Drag-to-rotate and scroll-to-zoom out of the box via `<GlyphOrbitControls>`.

## Try a Platonic solid

<Tabs syncKey="fw">
  <TabItem label="React">

```tsx
import { GlyphCamera, GlyphScene, GlyphOrbitControls, GlyphMesh } from "@glyphcss/react";
import { icosahedronPolygons } from "@glyphcss/core";

const icosa = icosahedronPolygons({ center: [0, 0, 0], size: 1, color: "#44ffcc" });

export function IcosahedronDemo() {
  return (
    <GlyphCamera rotX={0.4} zoom={0.32}>
      <GlyphScene mode="solid">
        <GlyphOrbitControls />
        <GlyphMesh polygons={icosa} />
      </GlyphScene>
    </GlyphCamera>
  );
}
```

  </TabItem>
  <TabItem label="Vue">

```vue
<template>
  <GlyphCamera :rot-x="0.4" :zoom="0.32">
    <GlyphScene mode="solid">
      <GlyphOrbitControls />
      <GlyphMesh :polygons="icosa" />
    </GlyphScene>
  </GlyphCamera>
</template>

<script setup lang="ts">
import { GlyphCamera, GlyphScene, GlyphOrbitControls, GlyphMesh } from "@glyphcss/vue";
import { icosahedronPolygons } from "@glyphcss/core";

const icosa = icosahedronPolygons({ center: [0, 0, 0], size: 1, color: "#44ffcc" });
</script>
```

  </TabItem>
  <TabItem label="Vanilla JS">

```ts
import { createGlyphCamera, createGlyphScene } from "glyphcss";
import { icosahedronPolygons } from "@glyphcss/core";

const host = document.querySelector<HTMLElement>("#scene")!;
const camera = createGlyphCamera({ rotX: 0.4, zoom: 0.32 });
const scene = createGlyphScene(host, { camera, mode: "solid", cols: 80, rows: 24 });
scene.add(icosahedronPolygons({ center: [0, 0, 0], size: 1, color: "#44ffcc" }));
```
